## Usage

```tsx
import { Tabs, TabsContent, TabsList, TabsTrigger } from '@tszhong0411/ui'
```

```tsx
<Tabs defaultValue='apple'>
  <TabsList>
    <TabsTrigger value='apple'>Apple</TabsTrigger>
    <TabsTrigger value='orange'>Orange</TabsTrigger>
    <TabsTrigger value='mango'>Mango</TabsTrigger>
  </TabsList>
  <TabsContent value='apple'>Apple</TabsContent>
  <TabsContent value='orange'>Orange</TabsContent>
  <TabsContent value='mango'>Mango</TabsContent>
</Tabs>
```
